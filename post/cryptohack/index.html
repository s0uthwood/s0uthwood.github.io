Encoding Challenge

Pwntools的简单使用

from pwn import *
import json

r = remote('socket.cryptohack.org', 13377)

def json_recv():
    line = r.recvline()
    return json.loads(line.decode())

def json_send(hsh):
    request = json.dumps(hsh).encode()
    r.sendline(request)

ENCODINGS = [
    "base64",
    "hex",
    "rot13",
    "bigint",
    "utf-8",
]

from base64 import b64decode
import codecs
from libnum import n2s

for _ in range(100):
	received = json_recv()

	print("Received type: ")
	print(received["type"])
	print("Received encoded value: ")
	print(received["encoded"])
	i = ENCODINGS.index(received['type'])
	decoded = received['encoded']
	if i == 0:
		decoded = b64decode(decoded)
		decoded = str(decoded)[2:-1]
	elif i == 1:
		decoded = n2s(int(decoded, 16))
		decoded = str(decoded)[2:-1]
	elif i == 2:
		decoded = codecs.decode(decoded, 'rot_13')
	elif i == 3:
		decoded = n2s(int(decoded, 16))
		decoded = str(decoded)[2:-1]
	elif i == 4:
		decoded = ''
		for c in received['encoded']:
			decoded += chr(c)
	print (decoded)
	to_send = {
	    "decoded": decoded
	}
	json_send(to_send)

print(json_recv())
