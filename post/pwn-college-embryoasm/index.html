from pwn import *
context(os='linux', arch='amd64', log_level='debug')
p = process('./embryoasm_level23')
p.recvuntil(b'bytes): \n')
p.send(asm('''
  mov rbp, rsp
  sub rsp, 0x500
  xor rbx, rbx
  xor rcx, rcx
judge1:
  cmp rcx, rsi
  jge finish1
  movb dl, byte ptr [rdi + rcx]
  mov eax, dword ptr -0x400[rbp + 4 * rdx]
  inc eax
  mov dword ptr -0x400[rbp + 4 * rdx], eax
  inc rcx
  jmp judge1
finish1:
  xor rbx, rbx
  xor rcx, rcx
  xor rax, rax
  xor rdx, rdx
judge2:
  cmp rcx, 0xff
  jg finish2
  mov ebx, dword ptr -0x400[rbp + 4 * rcx]
  cmp edx, ebx
  jge next
  mov edx, ebx
  mov rax, rcx
next:
  inc rcx
  jmp judge2
finish2:
  mov rsp, rbp
  ret
'''))
p.interactive()
